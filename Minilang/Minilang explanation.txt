function calloutBox(title, content, fill = C.blue4) {
  const borderDef = { style: BorderStyle.SINGLE, size: 6, color: C.blue2 };
  const rows = [];
  if (title) {
    rows.push(new TableRow({
      children: [new TableCell({
        shading: { fill: C.blue2, type: ShadingType.CLEAR },
        margins: { top: 80, bottom: 80, left: 160, right: 160 },
        borders: { top: borderDef, bottom: borderDef, left: borderDef, right: borderDef },
        children: [new Paragraph({
          children: [new TextRun({ text: title, bold: true, size: 20, color: C.white, font: "Arial" })]
        })]
      })]
    }));
  }
  rows.push(new TableRow({
    children: [new TableCell({
      shading: { fill, type: ShadingType.CLEAR },
      margins: { top: 120, bottom: 120, left: 200, right: 200 },
      borders: { top: borderDef, bottom: borderDef, left: borderDef, right: borderDef },
      children: content
    })]
  }));
  return new Table({
    width: { size: 9360, type: WidthType.DXA },
    columnWidths: [9360],
    rows,
  });
}

function twoColTable(headers, rows_data, widths = [2200, 7160]) {
  const headerBorder = { style: BorderStyle.SINGLE, size: 4, color: C.blue2 };
  const cellBorder   = { style: BorderStyle.SINGLE, size: 1, color: C.gray3  };

  function makeCell(text, isHeader = false, w = widths[0]) {
    return new TableCell({
      width: { size: w, type: WidthType.DXA },
      shading: { fill: isHeader ? C.blue : C.white, type: ShadingType.CLEAR },
      margins: { top: 80, bottom: 80, left: 120, right: 120 },
      borders: { top: isHeader ? headerBorder : cellBorder, bottom: cellBorder, left: cellBorder, right: cellBorder },
      verticalAlign: VerticalAlign.TOP,
      children: [new Paragraph({
        children: [new TextRun({
          text, size: 20, font: "Arial",
          color: isHeader ? C.white : C.gray,
          bold: isHeader,
        })]
      })]
    });
  }

  const tableRows = [];
  if (headers && headers.length) {
    tableRows.push(new TableRow({
      children: headers.map((h, i) => makeCell(h, true, widths[i]))
    }));
  }
  for (const row of rows_data) {
    tableRows.push(new TableRow({
      children: row.map((cell, i) => makeCell(cell, false, widths[i]))
    }));
  }
  return new Table({
    width: { size: 9360, type: WidthType.DXA },
    columnWidths: widths,
    rows: tableRows,
  });
}
